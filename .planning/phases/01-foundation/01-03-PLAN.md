```sql
-- Hybrid search: combines pgvector cosine similarity with PostgreSQL full-text search.
-- Returns chunks ranked by a weighted combination of both scores.
-- Parameters:
--   query_embedding: the query embedded as a vector
--   query_text: the raw query string for FTS
--   match_count: how many results to return (default 10)
--   date_after: filter sources published after this date (NULL = no filter)
--   date_before: filter sources published before this date (NULL = no filter)
CREATE OR REPLACE FUNCTION hybrid_search(
  query_embedding   vector(1536),
  query_text        text,
  match_count       int     DEFAULT 10,
  date_after        date    DEFAULT NULL,
  date_before       date    DEFAULT NULL
)
RETURNS TABLE (
  chunk_id      uuid,
  source_id     uuid,
  content       text,
  vector_score  float,
  fts_score     float,
  hybrid_score  float,
  title         text,
  author        text,
  url           text,
  published_at  timestamptz
)
LANGUAGE sql STABLE
AS $$
  SELECT
    c.id                                                          AS chunk_id,
    c.source_id,
    c.content,
    1 - (c.embedding <=> query_embedding)                        AS vector_score,
    ts_rank(to_tsvector('english', c.content),
            plainto_tsquery('english', query_text))              AS fts_score,
    -- Hybrid: 70% vector, 30% FTS (weights tunable here)
    0.7 * (1 - (c.embedding <=> query_embedding))
    + 0.3 * ts_rank(to_tsvector('english', c.content),
                    plainto_tsquery('english', query_text))      AS hybrid_score,
    s.title,
    s.author,
    s.url,
    s.published_at
  FROM chunks c
  JOIN sources s ON s.id = c.source_id
  WHERE
    c.embedding IS NOT NULL
    AND (date_after  IS NULL OR s.published_at >= date_after)
    AND (date_before IS NULL OR s.published_at <= date_before)
  ORDER BY hybrid_score DESC
  LIMIT match_count;
$$;
```
