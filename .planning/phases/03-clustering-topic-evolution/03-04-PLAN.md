```python
# Lazy import (existing pattern in cli.py)
from second_brain.ingestion.clustering import assign_topics_to_source, regenerate_topic_summary

# After Step 3 loop, inside Step 4:
changed_topic_ids: set[str] = set()
for source_id in newly_stored_source_ids:  # collect IDs during Step 3
    topic_id = assign_topics_to_source(source_id, db, embed_provider, llm_provider)
    if topic_id:
        changed_topic_ids.add(topic_id)

# Step 5: regenerate once per changed topic (NOT once per article)
for topic_id in changed_topic_ids:
    summary = regenerate_topic_summary(topic_id, db, llm_provider)
    db.table("topics").update({"summary": summary}).eq("id", topic_id).execute()
```
